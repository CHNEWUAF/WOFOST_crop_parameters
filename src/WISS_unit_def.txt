
public enum ScientificUnit {

    NODIM         (ScientificUnitRangeType.ALL         , "[-]"),                    // dimensionless

    DATEREL       (ScientificUnitRangeType.ALL         , "[Date relative]"),        // relative number of days
    HPA           (ScientificUnitRangeType.ZEROPOSITIVE, "[hpa]"),                  // hectopascal
    MBAR          (ScientificUnitRangeType.ZEROPOSITIVE, "[mbar]"),                 // millibar

    KG_HA         (ScientificUnitRangeType.ZEROPOSITIVE, "[kg.ha-1]"),              // kilogram per hectare
    KG_HA1HR1     (ScientificUnitRangeType.ALL,          "[kg.ha-1.hr-1]"),         // kilogram per hectare per hour
    KG_KG         (ScientificUnitRangeType.ZEROPOSITIVE, "[kg.kg-1]"),              // kilogram per kilogram
    KG_KG1D1      (ScientificUnitRangeType.ZEROPOSITIVE, "[kg.kg-1.d-1]"),          // kilogram per kilogram per day
    HA_HA         (ScientificUnitRangeType.ZEROPOSITIVE, "[ha.ha-1]"),              // hectare per hectare
    HA_HA1D1      (ScientificUnitRangeType.ZEROPOSITIVE, "[ha.ha-1.d-1]"),          // hectare per hectare per day
    HA_KG         (ScientificUnitRangeType.ZEROPOSITIVE, "[ha.kg-1]"),              // hectare per kilogram
    CELSIUS       (ScientificUnitRangeType.ALL         , "[C]"),                    // degrees Celsius
    CELSIUS_DAYS  (ScientificUnitRangeType.ZEROPOSITIVE, "[C.d]"),                  // Celsius degree days
    MJ_M2D1       (ScientificUnitRangeType.ZEROPOSITIVE, "[mj.m-2.d-1]"),           // megajoules per square meter per day

    MM            (ScientificUnitRangeType.ZEROPOSITIVE, "[mm]"),                   // millimeter
    MM_D1         (ScientificUnitRangeType.ZEROPOSITIVE, "[mm.d-1]"),               // millimeter per day
    CM            (ScientificUnitRangeType.ZEROPOSITIVE, "[cm]"),                   // centimeter
    CM_D1         (ScientificUnitRangeType.ZEROPOSITIVE, "[cm.d-1]"),               // centimeter per day
    M             (ScientificUnitRangeType.ZEROPOSITIVE, "[m]"),                    // meter
    M_S           (ScientificUnitRangeType.ZEROPOSITIVE, "[m.s-1]"),                // meter per second
    M_D1          (ScientificUnitRangeType.ZEROPOSITIVE, "[m.d-1]"),                // meter per day
    ANGULARDD     (ScientificUnitRangeType.ALL         , "[deg]"),                  // for latitude, longitude etc.
    HOUR          (ScientificUnitRangeType.ZEROPOSITIVE, "[hr]"),                   // decimal hours
    DAYS          (ScientificUnitRangeType.ZEROPOSITIVE, "[d]"),                    // number of days
    M2_M2         (ScientificUnitRangeType.ZEROPOSITIVE, "[m2.m-2]"),               // m2.m-2, for lai
    CM3_CM3       (ScientificUnitRangeType.ZEROPOSITIVE, "[cm3.cm-3]"),             // cm3.cm-3, for bulk density

    // crop parameter specific
    KG1CH20_KG1_D1          (ScientificUnitRangeType.ALL, "[kg.CH2O.kg-1.d-1]"),            // rel. maint. resp. rate stems
    KG_HA1HR1J1_M2S1        (ScientificUnitRangeType.ALL, "[kg.ha-1.hr-1.J-1.m2.s]"),       // light use efficiency

    X_C_Y_C1D1              (ScientificUnitRangeType.ALL, "[C;C.d]"),                       // e.g. for daily increase in temp. sum as function of av. temp.
    X_NODIM_Y_HA_KG         (ScientificUnitRangeType.ALL, "[-;ha.kg-1]"),                   // e.g. for leaf/stem area as function of DVS
    X_C_Y_KG_HA1HR1J1_M2S1  (ScientificUnitRangeType.ALL, "[C;kg.ha-1.hr-1.J-1.m2.s]"),     // e.g. for light-use efficiency as function of daily mean temp

    X_NODIM_Y_KG_KG         (ScientificUnitRangeType.ALL, "[-;kg.kg-1]"),                   // e.g. for fraction of total dry matter to roots as a function of DVS
    X_NODIM_Y_KG_KG1D1      (ScientificUnitRangeType.ALL, "[-;kg.kg-1.d-1]");               // e.g. for rel. death rate of roots as a function of DVS
